def somaL(lista):
    soma = 0
    for i in lista:
        soma = soma + i
    return soma
    
def mediaL(lista):
    for i in lista:
        soma = soma + lista[i]
    return mediaL

def mediaPonderada(lista, listaPesos):
    num = 0
    num2 = 0
    n = len(lista)

    for i in range(n):
        num += (lista[i] * listaPesos[i])
        num2 += listaPesos[i]

    medPonderada = num / num2
    return medPonderada
